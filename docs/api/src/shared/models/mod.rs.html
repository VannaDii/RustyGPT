43/// Represents a model available for chat completions.
44#[derive(Debug, Serialize, Deserialize, Clone)]
45pub struct Model {
46    /// The unique identifier of the model.
47    pub id: String,
48    /// Object type, fixed to "model" for `OpenAI` compatibility.
49    #[serde(default = "default_model_object")]
50    pub object: String,
51    /// Unix timestamp when the model metadata was created.
52    pub created: i64,
53    /// Owner identifier for the model.
54    pub owned_by: String,
55    /// Human-friendly display name (non-standard extension).
56    #[serde(skip_serializing_if = "Option::is_none")]
57    pub name: Option<String>,
58    /// Model category (non-standard extension).
59    #[serde(skip_serializing_if = "Option::is_none")]
60    pub model_type: Option<String>,
61}

70/// Request schema for `/v1/chat/completions`.
71#[derive(Debug, Serialize, Deserialize, Clone)]
72pub struct ChatCompletionRequest {
73    /// The model to use for the completion.
74    pub model: String,
75    /// The input messages for the chat.
76    pub messages: Vec<ChatCompletionMessage>,
77    /// Optional temperature for randomness in responses.
78    #[serde(default)]
79    pub temperature: Option<f32>,
80    /// Optional nucleus sampling parameter.
81    #[serde(default)]
82    pub top_p: Option<f32>,
83    /// Optional upper bound on generated tokens.
84    #[serde(default)]
85    pub max_tokens: Option<u32>,
86    /// Optional stop sequences (string or array of strings).
87    #[serde(default)]
88    pub stop: Option<Value>,
89    /// Optional presence penalty (ignored but reported).
90    #[serde(default)]
91    pub presence_penalty: Option<f32>,
92    /// Optional frequency penalty (ignored but reported).
93    #[serde(default)]
94    pub frequency_penalty: Option<f32>,
95    /// Optional end-user identifier.
96    #[serde(default)]
97    pub user: Option<String>,
98    /// Whether to stream deltas. Defaults to false.
99    #[serde(default)]
100    pub stream: Option<bool>,
101    /// Arbitrary metadata; `RustyGPT` extensions expect `metadata.rustygpt`.
102    #[serde(default)]
103    pub metadata: Option<Value>,
104}

106/// A single chat message supplied in the completion request or returned in a response.

108pub struct ChatCompletionMessage {

114    #[serde(skip_serializing_if = "Option::is_none")]

118/// Response schema for `/v1/chat/completions`.
119#[derive(Debug, Serialize, Deserialize, Clone)]
120pub struct ChatCompletionResponse {
121    /// Unique identifier for this completion.
122    pub id: String,
123    /// Object type, fixed to "chat.completion".
124    pub object: String,
125    /// Unix timestamp when the completion was created.
126    pub created: i64,
127    /// The model that generated the completion.
128    pub model: String,
129    /// The generated responses.
130    pub choices: Vec<ChatCompletionChoice>,
131    /// Token usage details.
132    #[serde(skip_serializing_if = "Option::is_none")]
133    pub usage: Option<UsageBreakdown>,
134    /// Optional fingerprint for model diagnostics.
135    #[serde(skip_serializing_if = "Option::is_none")]
136    pub system_fingerprint: Option<String>,
137    /// Warnings about ignored or adjusted parameters.
138    #[serde(default, skip_serializing_if = "Vec::is_empty")]
139    pub warnings: Vec<String>,
140}

142/// A single choice in the chat completion response.
143#[derive(Debug, Serialize, Deserialize, Clone)]
144pub struct ChatCompletionChoice {

157/// Streaming chunk response for `/v1/chat/completions?stream=true`.
158#[derive(Debug, Serialize, Deserialize, Clone)]
159pub struct ChatCompletionChunk {
160    /// Identifier matching the completion id.
161    pub id: String,
162    /// Object type, fixed to "chat.completion.chunk".
163    pub object: String,
164    /// Unix timestamp when the chunk was generated.
165    pub created: i64,
166    /// The model producing the chunk.
167    pub model: String,
168    /// Optional fingerprint for diagnostics.
169    #[serde(skip_serializing_if = "Option::is_none")]
170    pub system_fingerprint: Option<String>,
171    /// Token usage summary (present on terminal chunk).
172    #[serde(skip_serializing_if = "Option::is_none")]
173    pub usage: Option<UsageBreakdown>,
174    /// Chunk choices.
175    pub choices: Vec<ChatCompletionChunkChoice>,
176    /// Warnings propagated to the stream.
177    #[serde(default, skip_serializing_if = "Vec::is_empty")]
178    pub warnings: Vec<String>,
179}
180
181/// Choice entry within a streaming chunk.
182#[derive(Debug, Serialize, Deserialize, Clone)]
183pub struct ChatCompletionChunkChoice {
184    /// Choice index.
185    pub index: usize,
186    /// Delta describing incremental tokens.
187    pub delta: ChatCompletionChunkDelta,
188    /// Finish reason (set when `delta` is terminal).
189    #[serde(skip_serializing_if = "Option::is_none")]
190    pub finish_reason: Option<String>,
191}
192
193/// Delta payload for streaming chunks.
194#[derive(Debug, Serialize, Deserialize, Clone, Default)]
195pub struct ChatCompletionChunkDelta {
196    /// Optional role (present on first chunk).
197    #[serde(skip_serializing_if = "Option::is_none")]
198    pub role: Option<String>,
199    /// Partial content.
200    #[serde(skip_serializing_if = "Option::is_none")]
201    pub content: Option<String>,
202}
