7/// Builds ordered context slices for thread-aware completions.
8#[derive(Debug, Clone)]
9pub struct ThreadContextBuilder {
10    messages: Vec<MessageView>,
11}
12
13impl ThreadContextBuilder {
14    /// Creates a new builder with messages ordered by depth-first traversal.
15    #[must_use]
16    pub fn new(mut messages: Vec<MessageView>) -> Self {
17        messages.sort_by(|a, b| match a.depth.cmp(&b.depth) {
18            std::cmp::Ordering::Equal => a.path.cmp(&b.path),
19            ordering => ordering,
20        });
21        Self { messages }
22    }
23
24    /// Returns the root message (thread head) if present.
25    #[must_use]
26    pub fn root(&self) -> Option<&MessageView> {
27        self.messages.iter().find(|msg| msg.id == msg.root_id)
28    }
29
30    /// Returns the ancestor chain for the provided parent message identifier.
31    /// The chain is ordered from root to the parent inclusive.
32    #[must_use]
33    pub fn ancestor_chain(&self, parent_id: Uuid) -> Vec<MessageView> {
34        let Some(parent) = self.messages.iter().find(|msg| msg.id == parent_id) else {
35            return Vec::new();
36        };
37
38        let prefixes = path_prefix_set(&parent.path);
39
40        self.messages
41            .iter()
42            .filter(|msg| prefixes.contains(&msg.path))
43            .cloned()
44            .collect()
45    }
46
47    /// Returns the full thread in depth-first order, truncated optionally by depth.
48    #[must_use]
49    pub fn full_thread(&self, max_depth: Option<i32>) -> Vec<MessageView> {
50        max_depth.map_or_else(
51            || self.messages.clone(),
52            |limit| {
53                self.messages
54                    .iter()
55                    .filter(|msg| msg.depth <= limit)
56                    .cloned()
57                    .collect()
58            },
59        )
60    }
61
62    /// Returns the immediate children of a given parent ordered by path.
63    #[must_use]
64    pub fn children(&self, parent_id: Uuid) -> Vec<MessageView> {
65        let Some(parent) = self.messages.iter().find(|msg| msg.id == parent_id) else {
66            return Vec::new();
67        };
68
69        self.messages
70            .iter()
71            .filter(|msg| msg.parent_id == Some(parent.id))
72            .cloned()
73            .collect()
74    }
75}
76
77fn path_prefix_set(path: &str) -> HashSet<String> {
78    let segments: Vec<&str> = path.split('.').collect();
79    let mut prefixes = HashSet::with_capacity(segments.len());
80    let mut current = String::new();
81    for (idx, segment) in segments.iter().enumerate() {
82        if idx == 0 {
83            current.clear();
84        } else {
85            current.push('.');
86        }
87        current.push_str(segment);
88        prefixes.insert(current.clone());
89    }
90    prefixes
91}

93#[cfg(test)]
94mod tests {
95    use super::*;
96    use chrono::Utc;
97
98    fn sample_message(id: Uuid, parent_id: Option<Uuid>, path: &str, depth: i32) -> MessageView {
99        MessageView {
100            id,
101            conversation_id: Uuid::new_v4(),
102            root_id: parent_id.unwrap_or(id),
103            parent_id,
104            author_user_id: None,
105            role: crate::models::chat::MessageRole::User,
106            content: String::new(),
107            path: path.to_string(),
108            depth,
109            created_at: crate::models::timestamp::Timestamp(Utc::now()),
110        }
111    }
112
113    #[test]
114    fn ancestor_chain_returns_ordered_messages() {
115        let root_id = Uuid::new_v4();
116        let child_id = Uuid::new_v4();
117
118        let root = sample_message(root_id, None, "mroot", 1);
119        let child = sample_message(child_id, Some(root_id), "mroot.mchild", 2);
120
121        let builder = ThreadContextBuilder::new(vec![child, root]);
122        let ancestors = builder.ancestor_chain(child_id);
123
124        assert_eq!(ancestors.len(), 2);
125        assert_eq!(ancestors[0].id, root_id);
126        assert_eq!(ancestors[1].id, child_id);
127    }
128
129    #[test]
130    fn children_returns_direct_descendants() {
131        let root_id = Uuid::new_v4();
132        let child_one = Uuid::new_v4();
133        let child_two = Uuid::new_v4();
134
135        let root = sample_message(root_id, None, "mroot", 1);
136        let child1 = sample_message(child_one, Some(root_id), "mroot.m1", 2);
137        let child2 = sample_message(child_two, Some(root_id), "mroot.m2", 2);
138
139        let builder = ThreadContextBuilder::new(vec![child1, child2, root]);
140        let children = builder.children(root_id);
141
142        assert_eq!(children.len(), 2);
143        assert!(children.iter().any(|msg| msg.id == child_one));
144        assert!(children.iter().any(|msg| msg.id == child_two));
145    }
