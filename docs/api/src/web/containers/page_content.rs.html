1use yew::{Children, Classes, Html, Properties, classes, function_component, html};
2
3#[derive(Properties, PartialEq)]
4pub struct PageContentProps {
5    // Main content
6    #[prop_or_default]
7    pub children: Children,
8
9    // Additional classes for the page container
10    #[prop_or_default]
11    pub class: Classes,
12
13    // Whether to add padding to the content
14    #[prop_or(true)]
15    pub padding: bool,
16
17    // Whether to use a card-like container with shadow and border
18    #[prop_or(true)]
19    pub container: bool,
20}
21
22/// Page content container component
23/// Provides consistent styling for page content across the app
24#[function_component(PageContent)]
25pub fn page_content(props: &PageContentProps) -> Html {
26    // Generate the appropriate classes based on props
27    let container_classes = classes!(
28        if props.container {
29            classes!(
30                "bg-base-100",
31                "rounded-box",
32                "shadow-sm",
33                "border",
34                "border-base-300"
35            )
36        } else {
37            Classes::new()
38        },
39        if props.padding { "p-4 md:p-6" } else { "" },
40        props.class.clone()
41    );
42
43    html! {
44        <div class={container_classes}>
45            {props.children.clone()}
46        </div>
47    }
48}
