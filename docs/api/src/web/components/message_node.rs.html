1use shared::models::{MessageRole, MessageView, Timestamp};
2use yew::{Callback, Html, Properties, classes, function_component, html};
3
4#[derive(Properties, PartialEq, Clone)]
5pub struct MessageNodeProps {
6    pub message: MessageView,
7    pub on_reply: Callback<MessageView>,
8}
9
10const fn role_classes(role: MessageRole) -> &'static str {
11    match role {
12        MessageRole::User => "bg-primary text-primary-content",
13        MessageRole::Assistant => "bg-base-200 text-base-content",
14        MessageRole::System => "bg-base-300 text-base-content",
15        MessageRole::Tool => "bg-neutral text-neutral-content",
16    }
17}
18
19const fn role_label(role: MessageRole) -> &'static str {
20    match role {
21        MessageRole::User => "User",
22        MessageRole::Assistant => "Assistant",
23        MessageRole::System => "System",
24        MessageRole::Tool => "Tool",
25    }
26}
27
28fn indent_style(depth: i32) -> String {
29    let level = f64::from(depth.saturating_sub(1));
30    let rem = level * 1.25;
31    format!("margin-left: {rem}rem;")
32}
33
34fn format_timestamp(timestamp: &Timestamp) -> String {
35    timestamp.0.format("%H:%M:%S").to_string()
36}
37
38#[function_component(MessageNode)]
39pub fn message_node(props: &MessageNodeProps) -> Html {
40    let message = props.message.clone();
41    let on_reply = props.on_reply.clone();
42
43    let reply_callback = Callback::from(move |_| {
44        on_reply.emit(message.clone());
45    });
46
47    let classes = classes!(
48        "rounded-xl",
49        "px-4",
50        "py-3",
51        "shadow-sm",
52        role_classes(props.message.role)
53    );
54    let style = indent_style(props.message.depth);
55
56    html! {
57        <div class="mb-3 space-y-1" style={style}>
58            <div class="flex items-center gap-2 text-xs text-base-content/70">
59                <span class="font-semibold">{ role_label(props.message.role) }</span>
60                <span>{ format_timestamp(&props.message.created_at) }</span>
61            </div>
62            <div class={classes}>
63                { props.message.content.clone() }
64            </div>
65            <div class="flex items-center gap-2 text-xs">
66                <button
67                    class="btn btn-ghost btn-xs"
68                    type="button"
69                    onclick={reply_callback}
70                >
71                    {"Reply"}
72                </button>
73            </div>
74        </div>
75    }
76}
