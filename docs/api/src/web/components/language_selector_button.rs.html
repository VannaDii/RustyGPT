1use yew::{Callback, Html, Properties, function_component, html};
2
3use crate::LanguageInfo;

5#[derive(Properties, PartialEq)]
6pub struct LanguageSelectorButtonProps {
7    pub is_active: bool,
8    pub info: LanguageInfo,
9    pub on_click: Callback<String>,
10}

12#[function_component(LanguageSelectorButton)]
13pub fn language_selector(props: &LanguageSelectorButtonProps) -> Html {
14    let info = &props.info;
15    let code = info.code.to_string();
16    let on_click = props.on_click.clone();
17    html! {
18        <li>
19            <a
20                class={if props.is_active { "active" } else { "" }}
21                onclick={move |event: yew::MouseEvent| {
22                    event.prevent_default();
23                    on_click.emit(code.clone());
24                }}>
25                <span>{props.info.flag}</span>
26                <span>{props.info.native_name}</span>
27            </a>
28        </li>
29    }
