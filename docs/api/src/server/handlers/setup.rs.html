17// Handler for checking if the system is setup
18#[utoipa::path(
19    get,
20    path = "/setup",
21    responses(
22        (status = 200, description = "Setup status retrieved successfully", body = SetupResponse),
23        (status = 500, description = "Internal server error", body = ErrorResponse)
24    ),
25    tag = "Setup"
26)]
27#[instrument(skip(state))]
28pub async fn get_setup(State(state): State<Arc<AppState>>) -> Response {
29    info!("Received setup check request");
30    match is_setup(state.pool.as_ref()).await {
31        Ok(is_setup) => {
32            info!("Setup check completed, is_setup: {}", is_setup);
33            (StatusCode::OK, Json(SetupResponse { is_setup })).into_response()
34        }
35        Err(err) => {
36            info!("Setup check failed: {}", err);
37            (
38                StatusCode::INTERNAL_SERVER_ERROR,
39                Json(ErrorResponse {
40                    message: "Failed to check setup".to_string(),
41                    details: Some(err.to_string()),
42                }),
43            )
44                .into_response()
45        }
46    }
47}

49// Handler for configuring the system the first time
50#[utoipa::path(
51    post,
52    path = "/setup",
53    request_body = SetupRequest,
54    responses(
55        (status = 200, description = "Setup completed successfully"),
56        (status = 400, description = "Setup rejected by database", body = ErrorResponse),
57        (status = 500, description = "Internal server error", body = ErrorResponse)
58    ),
59    tag = "Setup"
60)]
61#[instrument(skip(state))]
62pub async fn post_setup(
63    State(state): State<Arc<AppState>>,
64    Json(payload): Json<SetupRequest>,
65) -> Response {
66    info!("Received setup request: {:?}", payload);
67    match init_setup(state.pool.as_ref(), &payload).await {
68        Ok(true) => {
69            info!("Setup completed successfully");
70            (StatusCode::OK).into_response()
71        }
72        Ok(false) => {
73            info!("Setup rejected by database");
74            (
75                StatusCode::BAD_REQUEST,
76                Json(ErrorResponse {
77                    message: "Database rejected setup.".to_string(),
78                    details: Some("`init_setup` returned false".to_string()),
79                }),
80            )
81                .into_response()
82        }
83        Err(err) => {
84            info!("Setup failed: {}", err);
85            (
86                StatusCode::INTERNAL_SERVER_ERROR,
87                Json(ErrorResponse {
88                    message: "Failed to setup".to_string(),
89                    details: Some(err.to_string()),
90                }),
91            )
92                .into_response()
93        }
94    }
95}

97// Function to register the setup routes
98#[instrument]
99pub fn setup_routes() -> Router<Arc<AppState>> {
100    info!("Registering setup routes");
101    Router::new().route("/setup", get(get_setup).post(post_setup))
