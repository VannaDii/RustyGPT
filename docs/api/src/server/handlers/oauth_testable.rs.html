1/// Testable handler functions that accept OAuth service as dependency
2use crate::{app_state::AppState, services::oauth_service_trait::OAuthService};
3use axum::{
4    extract::{Json, Query, State},
5    response::{IntoResponse, Redirect, Response},
6};
7use http::StatusCode;
8use shared::models::oauth::{OAuthCallback, OAuthRequest};
9use std::sync::Arc;
10
11/// Testable GitHub OAuth callback handler
12///
13/// # Arguments
14/// * `params` - OAuth callback parameters containing authorization code
15/// * `state` - Application state containing database pool
16/// * `oauth_service` - OAuth service implementation (can be mocked for testing)
17///
18/// # Returns
19/// Redirect response to success page or error response
20pub async fn github_oauth_callback_with_service<T: OAuthService>(
21    Query(params): Query<OAuthCallback>,
22    State(state): State<Arc<AppState>>,
23    oauth_service: T,
24) -> Response {
25    oauth_service
26        .handle_github_oauth(state.pool.as_ref(), params.code)
27        .await
28        .map_or_else(
29            |_| (StatusCode::INTERNAL_SERVER_ERROR, "OAuth failed").into_response(),
30            |user_id| {
31                Redirect::to(&format!("/auth-success.html?user_id={user_id}")).into_response()
32            },
33        )
34}

36/// Testable GitHub OAuth manual handler
37///
38/// # Arguments
39/// * `state` - Application state containing database pool
40/// * `payload` - OAuth request containing authorization code
41/// * `oauth_service` - OAuth service implementation (can be mocked for testing)
42///
43/// # Returns
44/// Success message with user ID or error response
45pub async fn github_oauth_manual_with_service<T: OAuthService>(
46    State(state): State<Arc<AppState>>,
47    Json(payload): Json<OAuthRequest>,
48    oauth_service: T,
49) -> Response {
50    oauth_service
51        .handle_github_oauth(state.pool.as_ref(), payload.auth_code)
52        .await
53        .map_or_else(
54            |_| (StatusCode::INTERNAL_SERVER_ERROR, "OAuth failed").into_response(),
55            |user_id| {
56                (
57                    StatusCode::OK,
58                    format!("GitHub OAuth successful, user_id: {user_id}"),
59                )
60                    .into_response()
61            },
62        )
63}
