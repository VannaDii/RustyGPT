20// Handler for initiating GitHub OAuth flow
21#[utoipa::path(
22    get,
23    path = "/oauth/github",
24    responses(
25        (status = 200, description = "Authorization URL retrieved", body = OAuthInitResponse),
26        (status = 500, description = "Internal server error", body = ErrorResponse)
27    ),
28    tag = "Auth"
29)]
30pub async fn github_oauth_init() -> Json<OAuthInitResponse> {
31    // In a real implementation, this would generate a proper OAuth URL with state
32    let github_client_id = env::var("GITHUB_CLIENT_ID").unwrap_or_default();
33    let redirect_uri = env::var("GITHUB_REDIRECT_URI").unwrap_or_default();
34    let auth_base_url = env::var("GITHUB_AUTH_URL").unwrap_or_default();
35
36    let auth_url = format!(
37        "{auth_base_url}?client_id={github_client_id}&redirect_uri={redirect_uri}&scope=user"
38    );
39
40    Json(OAuthInitResponse { auth_url })
41}
