18#[derive(Clone, Debug, Default)]
19pub struct RequestContext {

21    pub session: Option<SessionUser>,
22}

24impl RequestContext {

30#[derive(Clone)]
31pub struct RequestIdState {
32    header: HeaderName,
33}
34
35impl RequestIdState {
36    pub fn from_config(config: &Config) -> Self {
37        let header = HeaderName::from_str(&config.server.request_id_header)
38            .unwrap_or_else(|_| HeaderName::from_static("x-request-id"));
39        Self { header }
40    }
41}
42
43pub async fn assign_request_id(
44    State(state): State<RequestIdState>,
45    mut request: Request<Body>,
46    next: Next,
47) -> AppResult<Response> {
48    let header_name = state.header.clone();
49    let current = extract_request_id(request.headers(), &header_name);
50
51    let request_id = current.unwrap_or_else(|| Uuid::new_v4().to_string());
52
53    request.extensions_mut().insert(RequestContext {
54        request_id: request_id.clone(),
55        session: None,
56    });
57
58    request.headers_mut().insert(
59        header_name.clone(),
60        HeaderValue::from_str(&request_id)
61            .map_err(|_| ApiError::internal_server_error("failed to encode request id"))?,
62    );
63
64    let mut response = next.run(request).await;
65    response.headers_mut().insert(
66        header_name,
67        HeaderValue::from_str(&request_id)
68            .map_err(|_| ApiError::internal_server_error("failed to encode request id"))?,
69    );
70
71    Ok(response)
72}
73
74fn extract_request_id(headers: &HeaderMap, header: &HeaderName) -> Option<String> {
75    headers
76        .get(header)
77        .and_then(|value| value.to_str().ok())
78        .map(|s| s.trim().to_string())
79        .filter(|s| !s.is_empty())
80}

131        let mut headers = HeaderMap::new();
132        let header = HeaderName::from_static("x-request-id");
133        assert!(extract_request_id(&headers, &header).is_none());
134
135        headers.insert(&header, HeaderValue::from_static("   "));
136        assert!(extract_request_id(&headers, &header).is_none());
137
138        headers.insert(&header, HeaderValue::from_static(" 12345 "));
139        assert_eq!(
140            extract_request_id(&headers, &header).as_deref(),
141            Some("12345")

150    async fn assign_request_id_generates_and_propagates_new_id() {
151        let state = RequestIdState::from_config(&Config::default_for_profile(Profile::Test));
152        let app = Router::new()
153            .route("/", get(request_id_handler))
154            .layer(axum::middleware::from_fn_with_state(
155                state.clone(),
156                assign_request_id,
157            ))
158            .with_state(state.clone());
159
160        let response = app
161            .oneshot(Request::builder().uri("/").body(Body::empty()).unwrap())
162            .await
163            .unwrap();
164
165        let headers = response.headers().clone();
166        let id = headers
167            .get(&state.header)
168            .and_then(|value| value.to_str().ok())
169            .unwrap()
170            .to_string();
171        assert!(!id.is_empty());
172
173        let (_, body) = response.into_parts();
174        let body = axum::body::to_bytes(body, 1024).await.unwrap();
175        assert_eq!(std::str::from_utf8(&body).unwrap(), id);
176    }
