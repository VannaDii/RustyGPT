20// Middleware to check if a user is authenticated
21#[allow(clippy::too_many_lines)] // Tracking: auth-middleware-refactor
22#[instrument(skip(next, req))]
23pub async fn auth_middleware(mut req: Request<Body>, next: Next) -> Result<Response, StatusCode> {
24    let Some(config) = req.extensions().get::<Arc<Config>>().cloned() else {
25        return Err(StatusCode::UNAUTHORIZED);
26    };
27
28    if !config.features.auth_v1 {
29        return Ok(next.run(req).await);
30    }
31
32    let state = req
33        .extensions()
34        .get::<Arc<AppState>>()
35        .cloned()
36        .ok_or(StatusCode::INTERNAL_SERVER_ERROR)?;
37    let session_service = state
38        .sessions
39        .clone()
40        .ok_or(StatusCode::INTERNAL_SERVER_ERROR)?;
41
42    let session_cookie_name = &config.session.session_cookie_name;
43    let Some(token) = extract_session_cookie(req.headers(), session_cookie_name) else {
44        return Ok(unauthorized_response());
45    };
46
47    let user_agent = req
48        .headers()
49        .get(header::USER_AGENT)
50        .and_then(|value| value.to_str().ok())
51        .map(ToString::to_string);
52
53    let forwarded_ip = req
54        .headers()
55        .get("x-forwarded-for")
56        .and_then(|value| value.to_str().ok())
57        .and_then(|raw| raw.split(',').next())
58        .map(str::trim)
59        .filter(|segment| !segment.is_empty())
60        .map(ToString::to_string);
61
62    let real_ip = req
63        .headers()
64        .get("x-real-ip")
65        .and_then(|value| value.to_str().ok())
66        .map(|value| value.trim().to_string())
67        .filter(|value| !value.is_empty());
68
69    let fingerprint = req
70        .headers()
71        .get("x-client-fingerprint")
72        .and_then(|value| value.to_str().ok())
73        .map(ToString::to_string);
74
75    let ip_source = forwarded_ip.or(real_ip);
76    let mut metadata = SessionMetadata::default()
77        .with_user_agent(user_agent)
78        .with_fingerprint(fingerprint);
79
80    if let Some(ip_value) = ip_source {
81        match ip_value.parse::<IpAddr>() {
82            Ok(parsed) => {
83                metadata = metadata.with_ip(Some(parsed));
84            }
85            Err(_) => {
86                metadata = metadata.with_ip_str(Some(ip_value));
87            }
88        }
89    }
90
91    let validation = match session_service.validate_session(&token, &metadata).await {
92        Ok(Some(value)) => value,
93        Ok(None) => return Ok(unauthorized_response()),
94        Err(err) => {
95            warn!(error = %err, "session validation failed");
96            return Ok(map_session_error(err));
97        }
98    };
99
100    let SessionValidation {
101        user,
102        bundle,
103        rotated,
104    } = validation;
105
106    let request_id = req
107        .extensions()
108        .get::<RequestContext>()
109        .map(|ctx| ctx.request_id.clone())
110        .or_else(|| {
111            let header = HeaderName::from_str(&config.server.request_id_header)
112                .unwrap_or_else(|_| HeaderName::from_static("x-request-id"));
113            req.headers()
114                .get(&header)
115                .and_then(|value| value.to_str().ok())
116                .map(ToString::to_string)
117        })
118        .unwrap_or_default();
119
120    if let Some(context) = req.extensions_mut().get_mut::<RequestContext>() {
121        context.session = Some(user.clone());
122    } else {
123        req.extensions_mut().insert(RequestContext {
124            request_id,
125            session: Some(user.clone()),
126        });
127    }
128
129    let mut response = next.run(req).await;
130    let rotation_header_set = bundle.as_ref().is_some_and(|bundle| {
131        if let Ok(value) = http::HeaderValue::from_str(&bundle.session_cookie.to_string()) {
132            response.headers_mut().append(header::SET_COOKIE, value);
133        }
134        if let Ok(value) = http::HeaderValue::from_str(&bundle.csrf_cookie.to_string()) {
135            response.headers_mut().append(header::SET_COOKIE, value);
136        }
137        response.headers_mut().insert(
138            header::HeaderName::from_static("x-session-rotated"),
139            http::HeaderValue::from_static("1"),
140        );
141        true
142    });
143
144    if rotated && !rotation_header_set {
145        response.headers_mut().insert(
146            header::HeaderName::from_static("x-session-rotated"),
147            http::HeaderValue::from_static("1"),
148        );
149    }
150
151    Ok(response)
152}

154fn extract_session_cookie(headers: &http::HeaderMap, name: &str) -> Option<String> {
155    let value = headers.get(header::COOKIE)?.to_str().ok()?;
156    Cookie::split_parse(value)
157        .flatten()
158        .find(|cookie| cookie.name() == name)
159        .map(|cookie| cookie.value().to_string())
160}

162fn unauthorized_response() -> Response {
163    unauthorized_response_with("session")
164}
165
166fn unauthorized_response_with(scheme: &'static str) -> Response {
167    http::Response::builder()
168        .status(StatusCode::UNAUTHORIZED)
169        .header(header::WWW_AUTHENTICATE, scheme)
170        .body(Body::empty())
171        .unwrap()
172}
173
174fn conflict_response() -> Response {
175    http::Response::builder()
176        .status(StatusCode::CONFLICT)
177        .body(Body::empty())
178        .unwrap()
179}
180
181fn locked_response() -> Response {
182    http::Response::builder()
183        .status(StatusCode::LOCKED)
184        .body(Body::empty())
185        .unwrap()
186}
187
188fn map_session_error(error: SessionError) -> Response {
189    match error {
190        SessionError::DisabledUser => locked_response(),
191        SessionError::RotationRequired => conflict_response(),
192        SessionError::SuspiciousActivity => unauthorized_response_with("refresh"),
193        SessionError::SessionExpired
194        | SessionError::AbsoluteExpired
195        | SessionError::InvalidCredentials => unauthorized_response(),
196        other => {
197            warn!(error = %other, "unexpected session error");
198            unauthorized_response()
199        }
200    }
201}
