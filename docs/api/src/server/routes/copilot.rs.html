3use crate::{
4    app_state::AppState,

10/// Configures the Copilot API routes.
11///
12/// # Returns
13/// A [`Router`](axum::Router) with the Copilot API routes.
14pub fn create_router_copilot() -> Router<Arc<AppState>> {
15    Router::new()
16        .route("/v1/models", get(get_models))
17        .route("/v1/chat/completions", post(post_chat_completions))
18}

122    fn test_server() -> TestServer {
123        let assistant: Arc<dyn AssistantRuntime> = Arc::new(StubAssistant::new(stub_chunks()));
124        let config = Arc::new(Config::default_for_profile(Profile::Test));
125        let hub: SharedStreamHub = Arc::new(StreamHub::new(16, None, None));
126        let context = RequestContext {
127            request_id: "req".into(),
128            session: None,
129        };
130
131        let app_state = Arc::new(AppState {
132            assistant: Some(assistant),
133            ..AppState::default()
134        });
135
136        let app = create_router_copilot()
137            .layer(Extension(app_state.clone()))
138            .layer(Extension(config))
139            .layer(Extension(context))
140            .layer(Extension(hub))
141            .with_state(app_state);
142
143        TestServer::new(app).expect("router server")
144    }
